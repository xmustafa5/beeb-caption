# Rider Wallet (Phase 5a) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Give the rider a complete wallet surface — balance, top-up (cash or saved card via MockGateway), card-on-file CRUD, and transaction history — wired to the live Beeb backend.

**Architecture:** Mirror the Phase 1–4 pattern: one `services/wallet.ts` owning all backend shapes/mappers, TanStack Query hooks owning caching + mutations with hierarchical keys, and a new `app/(wallet)/` route group for screens. No Zustand (wallet is server state). Entry from Profile rows.

**Tech Stack:** Expo Router, TanStack Query, axios (`lib/api.ts`), React Hook Form + zod (already deps), `formatIqd` (`lib/format-currency.ts`), i18n (en/ar).

**Verification note:** this project has NO unit-test runner (only `expo lint`). The per-task gate is **`npx tsc --noEmit` + `npx expo lint` clean**, plus **curl endpoint probes** for the service layer. The authenticated happy-path (real top-up/card add) requires a rider token and is flagged needs-live-test, not claimed verified — same posture as the Abriyah flow.

```ts
import { api } from '@/lib/api'

export interface Wallet {
  balanceIqd: number
}

export type TxType = 'trip_fare' | 'daily_fee' | 'topup' | 'refund' | 'cancellation_penalty'
export type TxStatus = 'pending' | 'succeeded' | 'failed' | 'reversed'

export interface Transaction {
  id: string
  txType: TxType
  status: TxStatus
  amountIqd: number
  createdAt: string
  tripId?: string
  failureReason?: string
}

export interface PaymentMethod {
  id: string
  methodType: string
  maskedLast4: string | null
  isDefault: boolean
}

interface BackendWallet {
  balance_iqd: number
}
interface BackendTransaction {
  id: string
  tx_type: TxType
  status: TxStatus
  amount_iqd: number
  created_at: string
  trip_id?: string | null
  failure_reason?: string | null
}
interface BackendPaymentMethod {
  id: string
  method_type: string
  masked_last4: string | null
  is_default: boolean
}

function toTransaction(b: BackendTransaction): Transaction {
  return {
    id: b.id,
    txType: b.tx_type,
    status: b.status,
    amountIqd: b.amount_iqd,
    createdAt: b.created_at,
    tripId: b.trip_id ?? undefined,
    failureReason: b.failure_reason ?? undefined,
  }
}

function toPaymentMethod(b: BackendPaymentMethod): PaymentMethod {
  return {
    id: b.id,
    methodType: b.method_type,
    maskedLast4: b.masked_last4,
    isDefault: b.is_default,
  }
}

/** Wallet balance. Auto-provisions on first call. */
export async function getWallet(): Promise<Wallet> {
  const { data } = await api.get<BackendWallet>('/api/me/wallet')
  return { balanceIqd: data.balance_iqd }
}

/**
 * Top up the wallet. With `paymentMethodId` → MockGateway charge then credit;
 * without → cash/admin credit. amount<=0 → 400; gateway reject → 402.
 */
export async function topUp(amountIqd: number, paymentMethodId?: string): Promise<Transaction> {
  const { data } = await api.post<BackendTransaction>('/api/me/wallet/topup', {
    amount_iqd: amountIqd,
    ...(paymentMethodId ? { payment_method_id: paymentMethodId } : {}),
  })
  return toTransaction(data)
}

/** Newest-first transaction ledger (limit/offset; bare array). */
export async function listTransactions(limit = 20, offset = 0): Promise<Transaction[]> {
  const { data } = await api.get<BackendTransaction[]>('/api/me/transactions', {
    params: { limit, offset },
  })
  return (data ?? []).map(toTransaction)
}

export async function listPaymentMethods(): Promise<PaymentMethod[]> {
  const { data } = await api.get<{ items?: BackendPaymentMethod[] } | BackendPaymentMethod[]>(
    '/api/me/payment-methods',
  )
  const items = Array.isArray(data) ? data : (data.items ?? [])
  return items.map(toPaymentMethod)
}

/** Tokenize a card. Only `masked_last4` is returned; the raw number is never stored. */
export async function addPaymentMethod(
  cardNumber: string,
  setAsDefault?: boolean,
): Promise<PaymentMethod> {
  const { data } = await api.post<BackendPaymentMethod>('/api/me/payment-methods', {
    card_number: cardNumber,
    method_type: 'card',
    ...(setAsDefault ? { set_as_default: true } : {}),
  })
  return toPaymentMethod(data)
}

export async function setDefaultPaymentMethod(id: string): Promise<void> {
  await api.put(`/api/me/payment-methods/${id}/default`)
}

export async function deletePaymentMethod(id: string): Promise<void> {
  await api.delete(`/api/me/payment-methods/${id}`)
}
```

```ts
import { useMemo } from 'react'
import { useInfiniteQuery } from '@tanstack/react-query'
import { listTransactions, type Transaction } from '@/services/wallet'

const PAGE = 20

export function useTransactions() {
  const query = useInfiniteQuery({
    queryKey: ['transactions'],
    queryFn: ({ pageParam = 0 }) => listTransactions(PAGE, pageParam),
    initialPageParam: 0,
    getNextPageParam: (lastPage, allPages) =>
      lastPage.length < PAGE ? undefined : allPages.length * PAGE,
  })
  const items = useMemo<Transaction[]>(
    () => query.data?.pages.flat() ?? [],
    [query.data],
  )
  return { ...query, items }
}
```

- [ ] **Step 3: Write `hooks/use-payment-methods.ts`**

```ts
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query'
import {
  listPaymentMethods,
  addPaymentMethod,
  setDefaultPaymentMethod,
  deletePaymentMethod,
  type PaymentMethod,
} from '@/services/wallet'

export function usePaymentMethods() {
  return useQuery<PaymentMethod[]>({
    queryKey: ['payment-methods'],
    queryFn: listPaymentMethods,
    staleTime: 60 * 1000,
  })
}

export function useAddPaymentMethod() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: ({ cardNumber, setAsDefault }: { cardNumber: string; setAsDefault?: boolean }) =>
      addPaymentMethod(cardNumber, setAsDefault),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['payment-methods'] }),
  })
}

export function useSetDefaultPaymentMethod() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (id: string) => setDefaultPaymentMethod(id),
    // Optimistic: flip is_default locally, revert on error.
    onMutate: async (id) => {
      await qc.cancelQueries({ queryKey: ['payment-methods'] })
      const prev = qc.getQueryData<PaymentMethod[]>(['payment-methods'])
      qc.setQueryData<PaymentMethod[]>(['payment-methods'], (old) =>
        (old ?? []).map((m) => ({ ...m, isDefault: m.id === id })),
      )
      return { prev }
    },
    onError: (_e, _id, ctx) => {
      if (ctx?.prev) qc.setQueryData(['payment-methods'], ctx.prev)
    },
    onSettled: () => qc.invalidateQueries({ queryKey: ['payment-methods'] }),
  })
}

export function useDeletePaymentMethod() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (id: string) => deletePaymentMethod(id),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['payment-methods'] }),
  })
}
```

```tsx
const CREDIT_TYPES = new Set(['topup', 'refund'])

  daily_fee: 'calendar-outline',
  topup: 'add-circle-outline',

    <View style={{
      flexDirection: I18nManager.isRTL ? 'row-reverse' : 'row',
      alignItems: 'center',
      gap: Spacing.md,
      paddingVertical: Spacing.md,
      paddingHorizontal: Spacing.lg,
    }}>
      <View style={{
        width: 40, height: 40, borderRadius: 12,
        backgroundColor: colors.surface,
        alignItems: 'center', justifyContent: 'center',
      }}>
        <Icon name={ICON_BY_TYPE[tx.txType]} size={20} color={colors.text} />
      </View>
      <View style={{ flex: 1 }}>
        <Text style={{ ...Typography['body-md'], color: colors.text }} numberOfLines={1}>
          {t(`wallet.tx.${tx.txType}`)}
        </Text>
        <Text style={{ ...Typography['caption-sm'], color: colors.subtle, fontStyle: 'normal' }}>

        </Text>
      </View>
      <Text style={{
        ...Typography['body-md'],
        color: amountColor,
        fontFamily: 'Poppins_600SemiBold',
        fontVariant: ['tabular-nums'],
        textDecorationLine: muted ? 'line-through' : 'none',
      }}>
        {sign}{formatIqd(tx.amountIqd)}
      </Text>
    </View>
  )
}
```

```tsx
import { useState } from 'react'
import { View, Text, TouchableOpacity, ScrollView, I18nManager, Alert } from 'react-native'
import { useTranslation } from 'react-i18next'
import { useRouter } from 'expo-router'
import { useMutation, useQueryClient } from '@tanstack/react-query'
import { useThemeColors } from '@/hooks/use-theme-colors'
import { Typography } from '@/constants/Typography'
import { Spacing } from '@/constants/Spacing'
import { Icon } from '@/components/ui/icon'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/forms/input'
import { FormError } from '@/components/forms/form-error'
import { formatIqd } from '@/lib/format-currency'
import { parseApiError } from '@/lib/api'
import { topUp } from '@/services/wallet'
import { usePaymentMethods } from '@/hooks/use-payment-methods'

const PRESETS = [5000, 10000, 25000]
// Sentinel for the "cash / pay later" choice (omit payment_method_id).
const CASH = 'cash'

export default function TopUpScreen() {
  const { t } = useTranslation()
  const colors = useThemeColors()
  const router = useRouter()
  const qc = useQueryClient()
  const methods = usePaymentMethods()

  const [amount, setAmount] = useState<number>(PRESETS[1])
  const [custom, setCustom] = useState('')
  const [source, setSource] = useState<string>(CASH)
  const [error, setError] = useState<string | null>(null)

  const effectiveAmount = custom.trim() ? Number(custom.replace(/\D/g, '')) : amount

  const mutation = useMutation({
    mutationFn: () => topUp(effectiveAmount, source === CASH ? undefined : source),
    onMutate: () => setError(null),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['wallet'] })
      qc.invalidateQueries({ queryKey: ['transactions'] })
      router.back()
    },
    onError: (err) => {
      const info = parseApiError(err)
      if (info.status === 400) setError(t('wallet.amountInvalid'))
      else Alert.alert(t('wallet.topUpFailed'), info.backendMessage ?? t('common.error'))
    },
  })

  const valid = Number.isFinite(effectiveAmount) && effectiveAmount > 0

  return (
    <ScrollView contentContainerStyle={{ padding: Spacing.lg, gap: Spacing.lg }}>
      {/* Amount chips */}
      <Text style={{ ...Typography['heading-sm'], color: colors.text }}>{t('wallet.amount')}</Text>
      <View style={{ flexDirection: I18nManager.isRTL ? 'row-reverse' : 'row', gap: Spacing.sm }}>
        {PRESETS.map((p) => {
          const active = !custom.trim() && amount === p
          return (
            <TouchableOpacity
              key={p}
              onPress={() => { setAmount(p); setCustom('') }}
              activeOpacity={0.85}
              style={{
                flex: 1,
                paddingVertical: Spacing.md,
                borderRadius: 12,
                borderCurve: 'continuous',
                alignItems: 'center',
                backgroundColor: active ? colors.tint : colors.surface,
              }}
            >
              <Text style={{
                ...Typography['caption-sm'],
                color: active ? colors.onTint : colors.text,
                fontFamily: 'Poppins_600SemiBold',
                fontStyle: 'normal',
                fontVariant: ['tabular-nums'],
              }}>
                {formatIqd(p)}
              </Text>
            </TouchableOpacity>
          )
        })}
      </View>

      <Input
        label={t('wallet.amount')}
        value={custom}
        onChangeText={setCustom}
        keyboardType="number-pad"
        placeholder="0"
        leading={<Icon name="cash-outline" size={20} color={colors.subtle} />}
      />

      {/* Pay with */}
      <Text style={{ ...Typography['heading-sm'], color: colors.text }}>{t('wallet.payWith')}</Text>
      <View style={{ gap: Spacing.sm }}>
        {(methods.data ?? []).map((m) => (
          <SourceRow
            key={m.id}
            active={source === m.id}
            label={t('wallet.cardLabel', { last4: m.maskedLast4 ?? '----' })}
            icon="card"
            colors={colors}
            onPress={() => setSource(m.id)}
          />
        ))}
        <SourceRow
          active={source === CASH}
          label={t('wallet.cash')}
          icon="cash"
          colors={colors}
          onPress={() => setSource(CASH)}
        />
      </View>

      <FormError message={error} />
      <Button
        label={`${t('wallet.topUp')} ${formatIqd(valid ? effectiveAmount : 0)}`}
        loading={mutation.isPending}
        disabled={!valid}
        onPress={() => mutation.mutate()}
      />
    </ScrollView>
  )
}

interface SourceRowProps {
  active: boolean
  label: string
  icon: 'card' | 'cash'
  colors: ReturnType<typeof useThemeColors>
  onPress: () => void
}

function SourceRow({ active, label, icon, colors, onPress }: SourceRowProps) {
  return (
    <TouchableOpacity
      onPress={onPress}
      activeOpacity={0.85}
      style={{
        flexDirection: I18nManager.isRTL ? 'row-reverse' : 'row',
        alignItems: 'center',
        gap: Spacing.md,
        padding: Spacing.lg,
        borderRadius: 14,
        borderCurve: 'continuous',
        borderWidth: 2,
        borderColor: active ? colors.tint : 'transparent',
        backgroundColor: active ? colors.tint + '22' : colors.surface,
      }}
    >
      <Icon name={icon} size={22} color={colors.text} />
      <Text style={{ ...Typography['body-md'], color: colors.text, flex: 1 }}>{label}</Text>
      {active && <Icon name="checkmark-circle" size={22} color={colors.tint} />}
    </TouchableOpacity>
  )
}
```

```tsx
import { useState } from 'react'
import { View, Text, ScrollView, TouchableOpacity, Modal, Pressable, KeyboardAvoidingView, Platform, ActivityIndicator, I18nManager, Alert } from 'react-native'
import { useForm, Controller } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { z } from 'zod'
import { useTranslation } from 'react-i18next'
import { useThemeColors } from '@/hooks/use-theme-colors'
import { Typography } from '@/constants/Typography'
import { Spacing } from '@/constants/Spacing'
import { Icon } from '@/components/ui/icon'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/forms/input'
import { FormError } from '@/components/forms/form-error'
import {
  usePaymentMethods,
  useAddPaymentMethod,
  useSetDefaultPaymentMethod,
  useDeletePaymentMethod,
} from '@/hooks/use-payment-methods'

const schema = z.object({
  cardNumber: z.string().regex(/^\d{13,19}$/, 'wallet.cardNumberInvalid'),
  setAsDefault: z.boolean().optional(),
})
type FormData = z.infer<typeof schema>

export default function PaymentMethodsScreen() {
  const { t } = useTranslation()
  const colors = useThemeColors()
  const methods = usePaymentMethods()
  const addMutation = useAddPaymentMethod()
  const setDefault = useSetDefaultPaymentMethod()
  const del = useDeletePaymentMethod()
  const [adding, setAdding] = useState(false)

  const { control, handleSubmit, reset, formState: { errors } } = useForm<FormData>({
    resolver: zodResolver(schema),
    defaultValues: { cardNumber: '', setAsDefault: false },
  })

  const onAdd = (v: FormData) => {
    addMutation.mutate(
      { cardNumber: v.cardNumber, setAsDefault: v.setAsDefault },
      { onSuccess: () => { setAdding(false); reset() } },
    )
  }

  const confirmDelete = (id: string) => {
    Alert.alert(t('wallet.deleteConfirm'), undefined, [
      { text: t('common.cancel'), style: 'cancel' },
      { text: t('wallet.delete'), style: 'destructive', onPress: () => del.mutate(id) },
    ])
  }

  return (
    <View style={{ flex: 1, backgroundColor: colors.background }}>
      <ScrollView contentContainerStyle={{ padding: Spacing.lg, gap: Spacing.md }}>
        {methods.isLoading ? (
          <ActivityIndicator color={colors.tint} />
        ) : (methods.data ?? []).length === 0 ? (
          <Text style={{ ...Typography.body, color: colors.subtle, textAlign: 'center', paddingVertical: Spacing.xl }}>
            {t('wallet.noCards')}
          </Text>
        ) : (
          (methods.data ?? []).map((m) => (
            <View
              key={m.id}
              style={{
                flexDirection: I18nManager.isRTL ? 'row-reverse' : 'row',
                alignItems: 'center',
                gap: Spacing.md,
                padding: Spacing.lg,
                borderRadius: 14,
                borderCurve: 'continuous',
                backgroundColor: colors.surface,
              }}
            >
              <Icon name="card" size={24} color={colors.text} />
              <View style={{ flex: 1 }}>
                <Text style={{ ...Typography['body-md'], color: colors.text, fontVariant: ['tabular-nums'] }}>
                  {t('wallet.cardLabel', { last4: m.maskedLast4 ?? '----' })}
                </Text>
                {m.isDefault && (
                  <Text style={{ ...Typography['caption-sm'], color: colors.success, fontStyle: 'normal' }}>
                    {t('wallet.default')}
                  </Text>
                )}
              </View>
              {!m.isDefault && (
                <TouchableOpacity onPress={() => setDefault.mutate(m.id)}>
                  <Text style={{ ...Typography['caption-sm'], color: colors.tint, fontStyle: 'normal', fontFamily: 'Poppins_600SemiBold' }}>
                    {t('wallet.setDefault')}
                  </Text>
                </TouchableOpacity>
              )}
              <TouchableOpacity onPress={() => confirmDelete(m.id)}>
                <Icon name="trash-outline" size={20} color={colors.destructive} />
              </TouchableOpacity>
            </View>
          ))
        )}

        <Button label={t('wallet.addCard')} variant="secondary" onPress={() => setAdding(true)} />
      </ScrollView>

      {/* Add-card modal */}
      <Modal visible={adding} transparent animationType="slide" onRequestClose={() => setAdding(false)}>
        <KeyboardAvoidingView behavior={Platform.OS === 'ios' ? 'padding' : undefined} style={{ flex: 1 }}>
          <Pressable style={{ flex: 1, backgroundColor: 'rgba(0,0,0,0.4)', justifyContent: 'flex-end' }} onPress={() => setAdding(false)}>
            <Pressable
              onPress={(e) => e.stopPropagation()}
              style={{
                backgroundColor: colors.card,
                borderTopLeftRadius: 28,
                borderTopRightRadius: 28,
                borderCurve: 'continuous',
                padding: Spacing.xl,
                paddingBottom: Spacing.xl * 2,
                gap: Spacing.lg,
              }}
            >
              <Text style={{ ...Typography['heading-md'], color: colors.text }}>{t('wallet.addCard')}</Text>
              <Controller
                control={control}
                name="cardNumber"
                render={({ field: { value, onChange } }) => (
                  <Input
                    label={t('wallet.cardNumber')}
                    value={value}
                    onChangeText={onChange}
                    keyboardType="number-pad"
                    autoFocus
                    leading={<Icon name="card-outline" size={20} color={colors.subtle} />}
                  />
                )}
              />
              <FormError message={errors.cardNumber ? t(errors.cardNumber.message ?? 'wallet.cardNumberInvalid') : null} />
              <Button label={t('common.save')} loading={addMutation.isPending} onPress={handleSubmit(onAdd)} />
            </Pressable>
          </Pressable>
        </KeyboardAvoidingView>
      </Modal>
    </View>
  )
}
```
